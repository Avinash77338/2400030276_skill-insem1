# 1. Initialize a Git repository
git init

# 2. Add files to staging
git add .

# 3. Commit changes with a message
git commit -m "Initial commit"

# 4. Create a branch feature-login
git branch feature-login

# 5. Switch to the new branch
git checkout feature-login

# 6. Make a small change in app.js and commit
git add app.js
git commit -m "Added login feature in app.js"

# 7. Switch back to main branch
git checkout main

# 8. Change style.css and commit
git add style.css
git commit -m "Updated styles in style.css"

# 9. Merge feature-login into main
git merge feature-login

# 10. Handle merge conflicts (if any)
# Edit conflicting files, then:
git add .
git commit -m "Resolved merge conflicts"

# 11. Use git diff to view changes
git diff

# 12. Undo changes using git reset and git checkout
git reset --hard
git checkout .

# 13. Tag the latest commit as v1.0
git tag v1.0
